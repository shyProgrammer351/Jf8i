8. 반복문(for)

for( 초기식; 조건식; 증감식 ) {
  조건식이 참이면 실행하는 명령문      
} 		for(초기식;조건식;증감식)      
  명령문 한줄;
  		초기식;
while( 조건식 ) {
  조건식이 참이면 실행하는 명령문
  증감식;
}

for   초기식, 조건식, 증감식으로 구성되고, 조건식이 참일 때마다 해당 명령문을 실행하는 반복문
초기식 → 조건식(참) → 명령문 → 증감식 → 조건식(참) → ... → 증감식 → 조건식(거짓) → for문 종료

public class Main {
  public static void main( String[] args ) {
    int factorial = 1;
    System.out.print( "10! = " );
    for( int num = 10; num > 1; num-- ) {
      System.out.print( num + " * " );
      factorial *= num;
    }
    System.out.print( "1 = " + factorial );
  }
}
