import java.util.Scanner; 
	 
public class Calendar 
{ 
	public static void main( String[] args ) { 
		Scanner scan = new Scanner( System.in ); 
		final int SUNDAY = 0, SATURDAY = 6;		 
		int firstDay = 0, lastDay = 0, dayOfWeek = 0; 
	 
		System.out.print( "일(0),월(1),화(2),수(3),목(4),금(5),토(6) 중에서 \n이번달 1일은 무슨요일인가요 : " ); 
		firstDay = scan.nextInt(); 
		System.out.print( "이번달 말일은 언제인가요 : " ); 
		lastDay = scan.nextInt(); 
 
		System.out.println( "   일   월   화   수   목   금   토" );	 
	 
		( dayOfWeek = SUNDAY; 										 
		                        dayOfWeek < firstDay;  
		                                             dayOfWeek ) { 
			System.out.print( "     " ); 
		} 
		 
		 ( int day = 1; 									 
		                  day  lastDay;  						 
		                                 day, dayOfWeek ) { 
			System.out.printf( "", day );	 
			if ( ( dayOfWeek % 7 ) == SATURDAY ) { 
				System.out.println(); 
			} 
		} 
	 
	} 
}
