알고리즘

프로그램 시작 
   시작요일 및 말일 입력 받기 
   1일 이전은 공백 출력 
   1일부터 말일까지 출력 
      일주일을 한줄에 출력 
      즉, 토요일 후 일요일은 새 줄에서 시작 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter08/Calendar.java

	import java.util.Scanner;

		 

	public class Calendar

	{

		// 프로그램 시작 

1		public static void main( String[] args ) {

			Scanner scan = new Scanner( System.in );

			final int SUNDAY = 0, SATURDAY = 6;

2			int firstDay = 0, lastDay = 0, dayOfWeek = 0;

		 

			// 시작요일 및 말일 입력 받기 

3			System.out.print( "일(0),월(1),화(2),수(3),목(4),금(5),토(6) 중에서 \n이번달 1일은 무슨요일인가요 : " );

4			firstDay = scan.nextInt();

5			System.out.print( "이번달 말일은 언제인가요 : " );

6			lastDay = scan.nextInt();

	 

			// 요일 출력

7			System.out.println( "   일   월   화   수   목   금   토" );

		 

			// 1일 이전은 공백 출력 

8			for( dayOfWeek = SUNDAY;

9			                        dayOfWeek < firstDay;

10			                                             dayOfWeek++ ) {

11				System.out.print( "     " );

			}

			 

			// 1일부터 말일까지 출력 

12			for ( int day = 1;

13			                  day <= lastDay;

14			                                 day++, dayOfWeek++ ) {

				// 일주일을 한줄에 출력 

15				System.out.printf( "%5d", day );

				// 즉, 토요일 후 일요일은 새 줄에서 시작 

16				if ( ( dayOfWeek % 7 ) == SATURDAY ) {

17					System.out.println();

				}

			}

		 

		// 프로그램 종료 

18		}

	}
